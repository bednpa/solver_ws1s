93 explicit TwoWayDict(const MapFwdType& fwdMap) :

193 std::pair<ConstIteratorFwd, bool> Insert(
194 const std::pair<Type1, Type2>& value)
195 {
196 auto resPair = fwdmap_.insert(value);
197 if (!(resPair.second))
198 { // in case there is already some forward mapping for given value
199 assert(false); // fail gracefully
200 }
201
202 if (!(bwdmap_.insert(std::make_pair(value.second, value.first)).second))
203 { // in case there is already some backward mapping for given value
204 SIMLIB_ERROR("backward mapping for "
205 << Convert::ToString(value.second)
206 << " already found: "
207 << Convert::ToString(bwdmap_.find(value.second)->second));
208
209 assert(false); // fail gracefully
210 }
211
212 return resPair;
213 }

215 __attribute__((unused)) TwoWayDict Union(
216 const TwoWayDict& rhs) const
217 {
218 TwoWayDict result = *this;
219
220 // copy all pairs
221 for (ConstIteratorFwd itRhs = rhs.begin_fwd(); itRhs != rhs.end_fwd(); ++itRhs)
222 {
223 if ((result.fwdmap_.find(itRhs->first) != result.fwdmap_.end()) ||
224 (result.bwdmap_.find(itRhs->second) != result.bwdmap_.end()))
225 { // in case the first or the second component is already in the dictionary
226 assert(false); // fail gracefully
227 }
228
229 result.Insert(*itRhs);
230 }
231
232 return result;
233 }
