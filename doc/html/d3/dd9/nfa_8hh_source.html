53struct State // new definition of state with added jump/skip information
54{
55 unsigned long state;
56 unsigned long jump;

61

286struct Move {
287 Symbol symbol{};
288 StateSet states_to;
289
290 Move() = default;
291 explicit Move(Symbol symbolOnTransition) : symbol(symbolOnTransition), states_to() {}
292 Move(Symbol symbolOnTransition, State states_to) :
293 symbol(symbolOnTransition), states_to{states_to} {}
294 Move(Symbol symbolOnTransition, const StateSet& states_to) :
295 symbol(symbolOnTransition), states_to(states_to) {}
296
297 inline bool operator<(const Move& rhs) const { return symbol < rhs.symbol; }
298 inline bool operator<=(const Move& rhs) const { return symbol <= rhs.symbol; }
299 inline bool operator>(const Move& rhs) const { return symbol > rhs.symbol; }
300 inline bool operator>=(const Move& rhs) const { return symbol >= rhs.symbol; }
301 inline bool operator==(const Move& rhs) const { return symbol == rhs.symbol; }
302 inline bool operator!=(const Move& rhs) const { return symbol != rhs.symbol; }
303};

600 { // {{{
601 if (transition_relation.empty()) {
602 return false;

604
605 const Moves& tl = get_moves_from(src);
606 if (tl.empty()) {
607 return false;
608 }

612 }

616 }
617
