57 Simlib::Util::BinaryRelation compute_fw_direct_simulation(const Nfa& aut) {
58 Simlib::ExplicitLTS LTSforSimulation;
59 Symbol maxSymbol = 0;
60 const size_t state_num = aut.states_number();
61
62 for (State stateFrom = 0; stateFrom < state_num; ++stateFrom) {
63 for (const Move &t : aut.get_moves_from(stateFrom)) {
64 for (State stateTo : t.states_to) {
65 LTSforSimulation.add_transition(stateFrom, t.symbol, stateTo);
66 }
67 if (t.symbol > maxSymbol) {
68 maxSymbol = t.symbol;
69 }
70 }
71 }
72
73 // final states cannot be simulated by nonfinal -> we add new selfloops over final states with new symbol in LTS
74 for (State finalState : aut.final_states) {
75 LTSforSimulation.add_transition(finalState, maxSymbol + 1, finalState);
76 }
77
78 LTSforSimulation.init();
79 return LTSforSimulation.compute_simulation();
80 }

82 Nfa reduce_size_by_simulation(const Nfa& aut, StateToStateMap &state_map) {
83 Nfa result;
84 const auto sim_relation = compute_relation(aut, StringMap{{"relation", "simulation"}, {"direction", "forward"}});
85
86 auto sim_relation_symmetric = sim_relation;
87 sim_relation_symmetric.restrict_to_symmetric();
88
89 // for State q, quot_proj[q] should be the representative state representing the symmetric class of states in simulation
90 std::vector<size_t> quot_proj;
91 sim_relation_symmetric.get_quotient_projection(quot_proj);
92
93 const size_t num_of_states = aut.states_number();
94
95 // map each state q of aut to the state of the reduced automaton representing the simulation class of q
96 for (State q = 0; q < num_of_states; ++q) {
97 const State qReprState = quot_proj[q];
98 if (state_map.count(qReprState) == 0) { // we need to map q's class to a new state in reducedAut
99 const State qClass = result.add_state();
100 state_map[qReprState] = qClass;
101 state_map[q] = qClass;
102 } else {
103 state_map[q] = state_map[qReprState];
104 }
105 }
106
107 for (State q = 0; q < num_of_states; ++q) {
108 const State q_class_state = state_map.at(q);
109
110 if (aut.has_initial(q)) { // if a symmetric class contains initial state, then the whole class should be initial
111 result.make_initial(q_class_state);
112 }
113
114 if (quot_proj[q] == q) { // we process only transitions starting from the representative state, this is enough for simulation
115 for (const auto &q_trans : aut.get_moves_from(q)) {
116 // representatives_of_states_to = representatives of q_trans.states_to
117 const StateSet representatives_of_states_to = [&]{
118 StateSet state_set;
119 for (auto s : q_trans.states_to) {
120 state_set.insert(quot_proj[s]);
121 }
122 return state_set;
123 }();
124
125 // get the class states of those representatives that are not simulated by another representative in representatives_of_states_to
126 StateSet representatives_class_states;
127 for (const State s : representatives_of_states_to) {
128 bool is_state_important = true; // if true, we need to keep the transition from q to s
129 for (const State p : representatives_of_states_to) {
130 if (s != p && sim_relation.get(s, p)) { // if p (different from s) simulates s
131 is_state_important = false; // as p simulates s, the transition from q to s is not important to keep, as it is subsumed in transition from q to p
132 break;
133 }
134 }
135 if (is_state_important) {
136 representatives_class_states.insert(state_map.at(s));
137 }
138 }
139
140 // add the transition 'q_class_state-q_trans.symbol->representatives_class_states' at the end of transition list of transitions starting from q_class_state
141 // as the q_trans.symbol should be largest symbol we saw (as we iterate trough getTransitionsFromState(q) which is ordered)
142 result.transition_relation[q_class_state].push_back(Move(q_trans.symbol, representatives_class_states));
143 }
144
145 if (aut.has_final(q)) { // if q is final, then all states in its class are final => we make q_class_state final
146 result.make_final(q_class_state);
147 }
148 }
149 }
150
151 return result;
152 }

160 StateBoolArray compute_reachability(const Nfa& nfa) {
161 std::vector<State> worklist{ nfa.initial_states.ToVector() };
162
163 StateBoolArray reachable(nfa.states_number(), false);
164 for (const State state: nfa.initial_states)
165 {
166 reachable.at(state) = true;
167 }
168
169 State state{};
170 while (!worklist.empty())
171 {
172 state = worklist.back();
173 worklist.pop_back();
174
175 for (const auto& state_transitions: nfa.transition_relation[state])
176 {
177 for (const State target_state: state_transitions.states_to)
178 {
179 if (!reachable.at(target_state))
180 {
181 worklist.push_back(target_state);
182 reachable.at(target_state) = true;
183 }
184 }
185 }
186 }
187
188 return reachable;
189 }

198 StateBoolArray compute_reachability(const Nfa& nfa, const StateBoolArray& states_to_consider) {
199 std::vector<State> worklist{};
200 StateBoolArray reachable(nfa.states_number(), false);
201 for (const State state: nfa.initial_states) {
202 if (states_to_consider[state]) {
203 worklist.push_back(state);
204 reachable.at(state) = true;
205 }
206 }
207
208 State state;
209 while (!worklist.empty()) {
210 state = worklist.back();
211 worklist.pop_back();
212
213 for (const auto& state_transitions: nfa.transition_relation[state]) {
214 for (const State target_state: state_transitions.states_to) {
215 if (states_to_consider[target_state] && !reachable[target_state]) {
216 worklist.push_back(target_state);
217 reachable[target_state] = true;
218 }
219 }
220 }
221 }
222
223 return reachable;
224 }

232 void add_trimmed_transitions(const Nfa& nfa, const StateToStateMap& original_to_new_states_map, Nfa& trimmed_aut) {
233 // For each reachable original state 's' (which means it is mapped to the state of trimmed automaton)...
234 for (const auto& original_state_mapping: original_to_new_states_map)
235 {
236 // ...add all transitions from 's' to some reachable state to the trimmed automaton.
237 for (const auto& state_transitions_with_symbol: nfa.transition_relation[original_state_mapping.first])
238 {
239 Move new_state_trans_with_symbol(state_transitions_with_symbol.symbol);
240 for (State old_state_to: state_transitions_with_symbol.states_to)
241 {
242 auto iter_to_new_state_to = original_to_new_states_map.find(old_state_to);
243 if (iter_to_new_state_to != original_to_new_states_map.end())
244 {
245 // We can push here, because we assume that new states follow the ordering of original states.
246 new_state_trans_with_symbol.states_to.push_back(iter_to_new_state_to->second);
247 }
248 }
249 if (!new_state_trans_with_symbol.states_to.empty()) {
250 trimmed_aut.transition_relation[original_state_mapping.second].push_back(new_state_trans_with_symbol);
251 }
252 }
253 }

262 Nfa create_trimmed_aut(const Nfa& nfa, const StateToStateMap& original_to_new_states_map) {
263
264 Nfa trimmed_aut{ original_to_new_states_map.size() };
265
266 for (const State old_initial_state: nfa.initial_states)
267 {
268 if (original_to_new_states_map.find(old_initial_state) != original_to_new_states_map.end())
269 {
270 // we can use push_back here, because initial_states is ordered + new states follow the ordering of the old states
271 trimmed_aut.initial_states.push_back(original_to_new_states_map.at(old_initial_state));
272 }
273 }
274 for (const State old_final_state: nfa.final_states)
275 {
276 if (original_to_new_states_map.find(old_final_state) != original_to_new_states_map.end())
277 {
278 // we can use push_back here, because final_states is ordered + new states follow the ordering of the old states
279 trimmed_aut.final_states.push_back(original_to_new_states_map.at(old_final_state));
280 }
281 }
282
283 add_trimmed_transitions(nfa, original_to_new_states_map, trimmed_aut);
284 return trimmed_aut;
285 }

294 const State num_of_states{nfa.states_number() };
295 for (State src_state{ 0 }; src_state < num_of_states; ++src_state) {
296 for (const auto& symbol_transitions: nfa.transition_relation[src_state]) {

359void Nfa::add_trans(State state_from, Symbol symbol, State state_to) {
360 // TODO: Define own exception.
361 if (!is_state(state_from) || !is_state(state_to)) {
362 throw std::out_of_range(std::to_string(state_from) + " or " + std::to_string(state_to) + " is not a state.");
363 }
364
365 auto& state_transitions{transition_relation[state_from] };
366
367 if (state_transitions.ToVector().empty()) {
368 state_transitions.push_back({ symbol, state_to});
369 } else if (state_transitions.ToVector().back().symbol < symbol) {
370 state_transitions.push_back({ symbol, state_to});
371 } else {
372 const auto symbol_transitions{ state_transitions.find(Move{symbol }) };
373 if (symbol_transitions != state_transitions.end()) {
374 // Add transition with symbolOnTransition already used on transitions from stateFrom.
375 symbol_transitions->states_to.insert(state_to);
376 } else {
377 // Add transition to a new Move struct with symbolOnTransition yet unused on transitions from stateFrom.
378 const Move new_symbol_transitions{symbol, state_to };
379 state_transitions.insert(new_symbol_transitions);
380 }
381 }

384void Nfa::add_trans(const State state_from, const Symbol symbol, const StateSet& states_to) {
385 // TODO: Define own exception.
386 if (!is_state(state_from)) {
387 throw std::out_of_range(std::to_string(state_from) + " is not a state.");
388 }
389 for (const State state_to: states_to) {
390 if (!is_state(state_to)) {
391 throw std::out_of_range(std::to_string(state_to) + " is not a state.");
392 }
393 }
394
395 auto& state_transitions{transition_relation[state_from] };
396 if (state_transitions.ToVector().empty()) {
397 state_transitions.push_back({ symbol, states_to });
398 } else if (state_transitions.ToVector().back().symbol < symbol) {
399 state_transitions.push_back({ symbol, states_to });
400 } else {
401 const auto symbol_transitions{ state_transitions.find(Move{symbol }) };
402 if (symbol_transitions != state_transitions.end()) {
403 // Add transitions with symbol already used on transitions from state_from.
404 union_to_left(symbol_transitions->states_to,states_to);
405 return;
406 } else {
407 // Add transitions to a new Move struct with symbol yet unused on transitions from state_from.
408 const Move new_symbol_transitions {Move{symbol, states_to }};
409 state_transitions.insert(new_symbol_transitions);
410 }
411 }

414void Nfa::remove_trans(State src, Symbol symb, State tgt) {
415 auto& state_transitions{transition_relation[src] };
416 if (state_transitions.ToVector().empty()) {
417 throw std::invalid_argument(
418 "Transition [" + std::to_string(src) + ", " + std::to_string(symb) + ", " +
419 std::to_string(tgt) + "] does not exist.");
420 } else if (state_transitions.ToVector().back().symbol < symb) {
421 throw std::invalid_argument(
422 "Transition [" + std::to_string(src) + ", " + std::to_string(symb) + ", " +
423 std::to_string(tgt) + "] does not exist.");
424 } else {
425 const auto symbol_transitions{ state_transitions.find(Move{symb }) };
426 if (symbol_transitions == state_transitions.end()) {
427 throw std::invalid_argument(
428 "Transition [" + std::to_string(src) + ", " + std::to_string(symb) + ", " +
429 std::to_string(tgt) + "] does not exist.");
430 } else {
431 symbol_transitions->states_to.remove(tgt);
432 if (symbol_transitions->states_to.empty()) {
433 transition_relation[src].remove(*symbol_transitions);
434 }
435 }
436 }

489StateSet Nfa::get_useful_states() const
490{
491 if (initial_states.empty() || final_states.empty()) { return StateSet{}; }
492
493 const Nfa digraph{get_one_letter_aut() }; // Compute reachability on directed graph.
494 // Compute reachability from the initial states and use the reachable states to compute the reachability from the final states.
495 const StateBoolArray useful_states_bool_array{ compute_reachability(revert(digraph), compute_reachability(digraph)) };
496
497 const size_t num_of_states{states_number() };
498 StateSet useful_states{};
499 for (State original_state{ 0 }; original_state < num_of_states; ++original_state) {
500 if (useful_states_bool_array[original_state]) {
501 // We can use push_back here, because we are always increasing the value of original_state (so useful_states
502 // will always be ordered).
503 useful_states.push_back(original_state);
504 }
505 }
506 return useful_states;
507}

523 }
524 }
525
526 // for every state found in rhs, check its presence in lhs_states

528 if (haskey(lhs_states, rhs_st)) { return false; }

530

541 return false;

551void Mata::Nfa::make_complete(
552 Nfa& aut,
553 const Alphabet& alphabet,
554 State sink_state)
555{
556 std::list<State> worklist(aut.initial_states.begin(),
557 aut.initial_states.end());
558 std::unordered_set<State> processed(aut.initial_states.begin(),
559 aut.initial_states.end());
560 worklist.push_back(sink_state);
561 processed.insert(sink_state);
562 while (!worklist.empty())
563 {
564 State state = *worklist.begin();
565 worklist.pop_front();
566
567 std::set<Symbol> used_symbols;
568 if (!aut.has_no_transitions())
569 {
570 for (const auto &symb_stateset: aut[state]) {
571 // TODO: Possibly fix insert.
572 used_symbols.insert(symb_stateset.symbol);
573
574 const StateSet &stateset = symb_stateset.states_to;
575 for (const auto &tgt_state: stateset) {
576 bool inserted;
577 tie(std::ignore, inserted) = processed.insert(tgt_state);
578 if (inserted) { worklist.push_back(tgt_state); }
579 }
580 }
581 }
582
583 auto unused_symbols = alphabet.get_complement(used_symbols);
584 for (Symbol symb : unused_symbols)
585 {
586 aut.add_trans(state, symb, sink_state);
587 }
588 }
589}

591Nfa Mata::Nfa::remove_epsilon(const Nfa& aut, Symbol epsilon)
592{
593 Nfa result;
594
595 result.clear_nfa();
596 result.increase_size(aut.states_number());
597
598 // cannot use multimap, because it can contain multiple occurrences of (a -> a), (a -> a)
599 std::unordered_map<State, StateSet> eps_closure;
600
601 // TODO: grossly inefficient
602 // first we compute the epsilon closure
603 const size_t num_of_states{aut.states_number() };
604 for (size_t i{ 0 }; i < num_of_states; ++i)
605 {
606 for (const auto& trans: aut[i])
607 { // initialize
608 const auto it_ins_pair = eps_closure.insert({i, {i}});
609 if (trans.symbol == epsilon)
610 {
611 StateSet& closure = it_ins_pair.first->second;
612 // TODO: Fix possibly insert to OrdVector. Create list already ordered, then merge (do not need to resize each time);
613 closure.insert(trans.states_to);
614 }
615 }
616 }
617
618 bool changed = true;
619 while (changed) { // compute the fixpoint
620 changed = false;
621 for (size_t i=0; i < num_of_states; ++i) {
622 const auto& state_transitions{ aut[i] };
623 const auto state_symbol_transitions{
624 state_transitions.find(Move{epsilon}) };
625 if (state_symbol_transitions != state_transitions.end()) {
626 StateSet &src_eps_cl = eps_closure[i];
627 for (const State tgt: state_symbol_transitions->states_to) {
628 const StateSet &tgt_eps_cl = eps_closure[tgt];
629 for (const State st: tgt_eps_cl) {
630 if (src_eps_cl.count(st) == 0) {
631 changed = true;
632 break;
633 }
634 }
635 src_eps_cl.insert(tgt_eps_cl);
636 }
637 }
638 }
639 }
640
641 // now we construct the automaton without epsilon transitions
642 result.initial_states.insert(aut.initial_states);
643 result.final_states.insert(aut.final_states);
644 State max_state{};
645 for (const auto& state_closure_pair : eps_closure) { // for every state
646 State src_state = state_closure_pair.first;
647 for (State eps_cl_state : state_closure_pair.second) { // for every state in its eps cl
648 if (aut.has_final(eps_cl_state)) result.make_final(src_state);
649 for (const auto& symb_set : aut[eps_cl_state]) {
650 if (symb_set.symbol == epsilon) continue;
651
652 // TODO: this could be done more efficiently if we had a better add_trans method
653 for (State tgt_state : symb_set.states_to) {
654 max_state = std::max(src_state, tgt_state);
655 if (result.states_number() < max_state) {
656 result.increase_size_for_state(max_state);
657 }
658 result.add_trans(src_state, symb_set.symbol, tgt_state);
659 }
660 }
661 }
662 }
663
664 return result;
665}

668Nfa Mata::Nfa::revert(const Nfa& aut) {
669 Nfa result;
670 result.clear_nfa();
671 const size_t num_of_states{aut.states_number() };
672 result.increase_size(num_of_states);
673
674 for (State sourceState{ 0 }; sourceState < num_of_states; ++sourceState) {
675 for (const Move &transition: aut.transition_relation[sourceState]) {
676 for (const State targetState: transition.states_to) {
677
678 if (targetState.jump == 1)
679 {
680 result.add_trans(targetState, transition.symbol, State{sourceState, 1});
681 }
682 else
683 {
684 State inner_state;
685 inner_state.state = result.add_state();
686 inner_state.jump = targetState.jump - 1;
687 result.add_trans(targetState, 0, State{inner_state, targetState.jump - 1});
688 result.add_trans(targetState, 1, State{inner_state, targetState.jump - 1});
689 result.add_trans(inner_state, transition.symbol, State{sourceState, 1});
690 }
691 }
692 }
693 }
694
695 result.initial_states = aut.final_states;
696 result.final_states = aut.initial_states;
697
698 return result;
699}

718bool Mata::Nfa::is_complete(const Nfa& aut, const Alphabet& alphabet)
719{
720 Util::OrdVector<Symbol> symbs_ls = alphabet.get_alphabet_symbols();
721 Util::OrdVector<Symbol> symbs(symbs_ls.cbegin(), symbs_ls.cend());
722
723 // TODO: make a general function for traversal over reachable states that can
724 // be shared by other functions?
725 std::list<State> worklist(aut.initial_states.begin(),
726 aut.initial_states.end());
727 std::unordered_set<State> processed(aut.initial_states.begin(),
728 aut.initial_states.end());
729
730 while (!worklist.empty())
731 {
732 State state = *worklist.begin();
733 worklist.pop_front();
734
735 size_t n = 0; // counter of symbols
736 if (!aut.has_no_transitions()) {
737 for (const auto &symb_stateset: aut[state]) {
738 ++n;
739 if (!haskey(symbs, symb_stateset.symbol)) {
740 throw std::runtime_error(std::to_string(__func__) +
741 ": encountered a symbol that is not in the provided alphabet");
742 }
743
744 const StateSet &stateset = symb_stateset.states_to;
745 for (const auto &tgt_state: stateset) {
746 bool inserted;
747 tie(std::ignore, inserted) = processed.insert(tgt_state);
748 if (inserted) { worklist.push_back(tgt_state); }
749 }
750 }
751 }
752
753 if (symbs.size() != n) { return false; }
754 }
755
756 return true;
757}

760{

762 {
763 return {{}, true};
764 }
765
766 Run word;

768 for (size_t i = 1; i < run.path.size(); ++i)
769 {
770 State newSt = run.path[i];

772

775 for (const auto &symbolMap: aut.transition_relation[cur]) {
776 for (State st: symbolMap.states_to) {
777 if (st == newSt) {

779 found = true;

782 }
783
784 if (found) { break; }
785 }
786 }
787
788 if (!found)
789 {
790 return {{}, false};
791 }
792
793 cur = newSt; // update current state
794 }
795
796 return {word, true};

1064Nfa Mata::Nfa::uni(const Nfa &lhs, const Nfa &rhs) {
1065 // pridavam levy automat do praveho automatu
1066 Nfa unionAutomaton = rhs;
1067
1068 // mapuju stav leveho automatu na jeho odpovidajici novy stav v pravem automatu
1069 StateToStateMap thisStateToUnionState;
1070 for (State thisState = 0; thisState < lhs.transition_relation.size(); ++thisState) {
1071 thisStateToUnionState[thisState] = unionAutomaton.add_state();
1072 }
1073
1074 // puvodni pocatecni stavy leveho autmatu nastavim jako pocateni v pravem automatu
1075 for (State thisInitialState : lhs.initial_states) {
1076 unionAutomaton.make_initial(thisStateToUnionState[thisInitialState]);
1077 }
1078
1079 // puvodni koncove stavy leveho autmatu nastavim jako pocateni v pravem automatu
1080 for (State thisFinalState : lhs.final_states) {
1081 unionAutomaton.make_final(thisStateToUnionState[thisFinalState]);
1082 }
1083
1084 // presun prechodu
1085 for (State thisState = 0; thisState < lhs.transition_relation.size(); ++thisState) {
1086 State unionState = thisStateToUnionState[thisState];
1087 for (const Move &transitionFromThisState : lhs.transition_relation[thisState]) {
1088
1089 Move transitionFromUnionState(transitionFromThisState.symbol, StateSet{});
1090
1091 for (State stateTo : transitionFromThisState.states_to) {
1092 transitionFromUnionState.states_to.insert(State{thisStateToUnionState[stateTo].state, stateTo.jump});
1093 }
1094
1095 unionAutomaton.transition_relation[unionState].push_back(transitionFromUnionState);
1096 }
1097 }
1098
1099 return unionAutomaton;
1100}

1150Nfa Mata::Nfa::determinize(
1151 const Nfa& const_aut,
1152 std::unordered_map<StateSet, State> *subset_map)
1153{
1154 Nfa aut = const_aut;
1155
1156 Nfa result;
1157 //assuming all sets states_to are non-empty
1158 std::vector<std::pair<State, StateSet>> worklist;
1159 bool deallocate_subset_map = false;
1160 if (subset_map == nullptr) {
1161 subset_map = new std::unordered_map<StateSet,State>();
1162 deallocate_subset_map = true;
1163 }
1164
1165 result.clear_nfa();
1166
1167 const StateSet S0 = Mata::Util::OrdVector<State>(aut.initial_states.ToVector());
1168 const State S0id = result.add_state();
1169 result.make_initial(S0id);
1170
1171 //for fast detection of a final state in a set.
1172 std::vector<bool> isFinal(aut.states_number(), false);
1173 for (const auto &q: aut.final_states) {
1174 isFinal[q] = true;
1175 }
1176
1177 if (contains_final(S0, isFinal)) {
1178 result.make_final(S0id);
1179 }
1180 worklist.emplace_back(std::make_pair(S0id, S0));
1181
1182 (*subset_map)[Mata::Util::OrdVector<State>(S0)] = S0id;
1183
1184 if (aut.has_no_transitions())
1185 return result;
1186
1187 using Iterator = Mata::Util::OrdVector<Move>::const_iterator;
1188 Mata::Util::SynchronizedExistentialIterator<Iterator> synchronized_iterator;
1189
1190 Nfa aut_copy = aut;
1191 std::vector<std::pair<State,State>> end_inner_states;
1192
1193 while (!worklist.empty()) {
1194 const auto Spair = worklist.back();
1195 worklist.pop_back();
1196 const StateSet S = Spair.second;
1197 const State Sid = Spair.first;
1198 if (S.empty()) {
1199 break; //this should not happen assuming all sets states_to are non empty
1200 }
1201
1202 // add moves of S to the sync ex iterator
1203 // TODO: shouldn't we also reset first?
1204 Moves act_moves[S.size()];
1205 int i = 0;
1206 for (State q: S) {
1207 act_moves[i] = aut_copy.transition_relation[q];
1208 i++;
1209 }
1210 i = 0;
1211 for (State q: S) {
1212 Mata::Util::push_back(synchronized_iterator, act_moves[i]);
1213 i++;
1214 }
1215
1216 while (synchronized_iterator.advance()) {
1217 // extract post from the sychronized_iterator iterator
1218 std::vector<Iterator> moves = synchronized_iterator.get_current();
1219 Symbol currentSymbol = (*moves.begin())->symbol;
1220 std::vector<State> old_T;
1221
1222 for (auto m: moves){
1223 for (auto s: m->states_to) {
1224 old_T.push_back(s);
1225 }
1226 }
1227
1228 unsigned long min_t_jump = std::min_element(old_T.begin(), old_T.end(), [](const auto &e1, const auto &e2) {return e1.jump < e2.jump; })->jump;
1229 StateSet T;
1230
1231 for (const auto &t: old_T) {
1232 if (min_t_jump != t.jump) {
1233 for (const auto &s: S) {
1234 if (aut_copy.has_trans(s, currentSymbol, State{t.state, t.jump})) {
1235
1236 // remove A --> B
1237 aut_copy.remove_trans(s, currentSymbol, State{t.state, t.jump});
1238
1239 // create inner state
1240 State inner_state;
1241 auto it = std::find_if(end_inner_states.begin(), end_inner_states.end(), [&t,&min_t_jump](const auto &pair) { return pair.first.state == t.state && pair.first.jump - pair.second.jump == t.jump - min_t_jump && t.jump - min_t_jump > 0; });
1242
1243 if (it != end_inner_states.end()) {
1244 inner_state.state = it->second.state;
1245 inner_state.jump = min_t_jump;
1246 } else {
1247 inner_state.state = aut_copy.add_state();
1248 inner_state.jump = min_t_jump;
1249 isFinal.insert(isFinal.begin() + inner_state, false);
1250 end_inner_states.push_back(std::pair(t, inner_state));
1251 // add X --> B
1252 aut_copy.add_trans(inner_state, 0, State{t.state, t.jump - min_t_jump});
1253 aut_copy.add_trans(inner_state, 1, State{t.state, t.jump - min_t_jump});
1254 }
1255
1256 // set B = X
1257 T.insert(State{inner_state, min_t_jump});
1258 // add A --> X
1259 aut_copy.add_trans(s, currentSymbol, State{inner_state, min_t_jump});
1260 }
1261 }
1262 } else {
1263 T.insert(State{t.state, min_t_jump});
1264 }
1265 }
1266
1267 const auto existingTitr = subset_map->find(T);
1268 State Tid; // new state
1269 if (existingTitr != subset_map->end()) {
1270 Tid = existingTitr->second;
1271 Tid.jump = min_t_jump;
1272 } else {
1273 Tid = result.add_state();
1274 Tid.jump = min_t_jump;
1275 (*subset_map)[Mata::Util::OrdVector<State>(T)] = Tid;
1276 if (contains_final(T, isFinal)) {
1277 result.make_final(Tid);
1278 }
1279 worklist.emplace_back(std::make_pair(Tid, T));
1280 }
1281 result.transition_relation[Sid].push_back(Move(currentSymbol, Tid));
1282 }
1283 }
1284
1285 if (deallocate_subset_map) { delete subset_map; }
1286
1287 return result;
1288}

1305 state_map = new StringToStateMap();

1311 if (!state_map->count(str)) {
1312 State state = aut.add_state();

1327 if (parsec.dict.end() != it)

1330 {

1334 }

1344 }
1345 }
1346
1347 for (const auto& body_line : parsec.body)

1349 if (body_line.size() != 3)

1351 // clean up
1352 clean_up();
