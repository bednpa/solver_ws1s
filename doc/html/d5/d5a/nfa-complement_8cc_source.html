25Nfa Mata::Nfa::Algorithms::complement_classical(
26 const Nfa& aut,
27 const Alphabet& alphabet,
28 std::unordered_map<StateSet, State>* subset_map)
29{ // {{{
30 Nfa result;
31
32 bool delete_subset_map = false;
33 if (nullptr == subset_map)
34 {
35 subset_map = new std::unordered_map<StateSet, State>();
36 delete_subset_map = true;
37 }
38
39 result = determinize(aut, subset_map);
40 State sink_state = result.states_number() + 1;
41 result.increase_size(sink_state+1);
42 assert(sink_state < result.states_number());
43 auto it_inserted_pair = subset_map->insert({{}, sink_state});
44 if (!it_inserted_pair.second)
45 {
46 sink_state = it_inserted_pair.first->second;
47 }
48
49 make_complete(result, alphabet, sink_state);
50 StateSet old_fs = std::move(result.final_states);
51 result.final_states = { };
52 assert(result.initial_states.size() == 1);
53
54 auto make_final_if_not_in_old = [&](const State& state) {
55 if (!haskey(old_fs, state))
56 {
57 result.final_states.insert(state);
58 }
59 };
60
61 make_final_if_not_in_old(*result.initial_states.begin());
62
63 for (const auto& trs : result) {
64 make_final_if_not_in_old(trs.tgt);
65 }
66
67 if (delete_subset_map)
68 {
69 delete subset_map;
70 }
71
72 return result;
73} // complement_classical }}}

89void Mata::Nfa::complement_in_place(Nfa& aut) {
90 StateSet newFinalStates;
91
92 const size_t size = aut.transition_relation.size();
93 for (State q = 0; q < size; ++q) {
94 if (aut.final_states.count(q) == 0) {
95 newFinalStates.insert(q);
96 }
97 }
98
99 aut.final_states = newFinalStates;
100}

102Nfa Mata::Nfa::complement(
103 const Nfa& aut,
104 const Alphabet& alphabet,
105 const StringMap& params,
106 std::unordered_map<StateSet, State> *subset_map)
107{
108 Nfa result;
109 // setting the default algorithm
110 decltype(Algorithms::complement_classical)* algo = Algorithms::complement_classical;
111 if (!haskey(params, "algo")) {
112 throw std::runtime_error(std::to_string(__func__) +
113 " requires setting the \"algo\" key in the \"params\" argument; "
114 "received: " + std::to_string(params));
115 }
116
117 const std::string& str_algo = params.at("algo");
118 if ("classical" == str_algo) { /* default */ }
119 else {
120 throw std::runtime_error(std::to_string(__func__) +
121 " received an unknown value of the \"algo\" key: " + str_algo);
122 }
123
124 return algo(aut, alphabet, subset_map);
125} // complement
