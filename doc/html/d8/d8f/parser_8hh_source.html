42struct ParsedSection
43{
44 std::string type;
45 KeyListStore dict;

55 bool empty() const { return type.empty() && dict.empty() && body.empty(); }

58 friend std::ostream& operator<<(std::ostream& os, const ParsedSection& parsec)
59 { // {{{
60 os << "@" << parsec.type << "\n";
61 for (const auto& string_list_pair : parsec.dict)
62 {
63 os << "%" << string_list_pair.first;
64 for (const std::string& str : string_list_pair.second)
65 {
66 os << " " << str;
67 }
68
69 os << "\n";
70 }
71
72 os << "# Body:\n";
73 for (const auto& body_line : parsec.body)
74 {
75 bool first = true;
76 for (const std::string& str : body_line)
77 {
78 if (!first) { os << " ";}
79 first = false;
80 os << str;
81 }
82 os << "\n";
83 }
84
85 return os;
86 } // operator<< }}}
87
89 bool operator==(const ParsedSection& rhs) const
90 { // {{{
91 return
92 this->type == rhs.type &&
93 this->dict == rhs.dict &&
94 this->body == rhs.body;
95 } // }}}
96 bool operator!=(const ParsedSection& rhs) const { return !(*this == rhs); }
97
99 const std::vector<std::string>& operator[](const std::string& key) const
100 { // {{{
101 auto it = this->dict.find(key);
102 if (this->dict.end() == it) {
103 assert(false);
104 // return this->dict.at("");
105 }
106
107 return it->second;
108 } // operator[] }}}
109
111 bool haskey(const std::string& key) const
112 {
113 return this->dict.end() != this->dict.find(key);
114 }
115};
