12 /* Two classes that provide "synchronized" iterators through a vector of ordered vectors,
13 * (or of some ordered OrdContainer that have a similar iterator),
14 * needed in computation of post
15 * in subset construction, product, and non-determinization.
16 * The Type stored in OrdContainers must be comparable with <,>,==,!=,<=,>=,
17 * and it must be a total (linear) ordering.
18 * The intended usage in, for instance, determinisation is for Type to be TransSymbolStates.
19 * TransSymbolStates is ordered by the symbol.
20 *
21 * SyncrhonisedIterator is the parent virtual class.
22 * It stores a vector of end-iterators for the OrdContainer v and a vector of current positions.
23 * They are filled in using the function push_back(begin,end), that adds begin and end iterators of v to positions and ends, respectively
24 * Method advance advances all positions forward so that they are synchronized on the next smallest equiv class
25 * (next smallest symbol in the case of TransSymbolStates).
26 *
27 * There are two versions of the class.
28 * i) In product, ALL positions must point to currently the smallest equiv. class (Moves with the same symbol).
29 * Method get_current then returns the vector of all position iterators, synchronized.
30 * ii) In determinization, it is enough that there EXISTS a position that points to the smallest class.
31 * Method get_current then returns the vector of only those positions that point to the smallest equiv. class.
32 *
33 * Usage: 0) construct, 1) fill in using push_back, iterate using advance and get_current, 2) reset, goto 1)
34 *
35 * The memory allocated internally for positions and ends is kept after reset, so it is advisable to use the same iterator for many iterations, as
36 * opposed to creating a new one for each iteration.
37 */
38 template<typename Iterator> class SynchronizedIterator {
39 public:
40
41 std::vector<Iterator> positions;
42 std::vector<Iterator> ends;
43
44 //@param size Number of elements to reserve up-front for positions and ends.
45 explicit SynchronizedIterator(const int size = 0)
46 {
47 positions.reserve(size);
48 ends.reserve(size);
49 };
50
51 /* This is supposed to be called only before an iteration,
52 * after constructor of reset.
53 * Calling after advance breaks the iterator.
54 * Specifies begin and end of one vector, to initialise before the iteration starts.
55 */
56 virtual void push_back (const Iterator &begin, const Iterator &end) {
57 // Btw, I don't know what I am doing with the const & parameter passing,
58 // begin is actually incremented in advance ...? But tests do pass ...
59 this->positions.emplace_back(begin);
60 this->ends.emplace_back(end);
61 };
62
63
64 /* Empties positions and ends.
65 * Though they should keep the allocated space.
66 * @param size Number of elements to reserve up-front for positions and ends.
67 */
68 void reset(const int size = 0) {
69 positions.clear();
70 ends.clear();
71 if (size > 0) {
72 positions.reserve(size);
73 ends.reserve(size);
74 }
75 };
76
77 virtual bool advance() = 0;
78 virtual const std::vector<Iterator> get_current() = 0;
79 };

83 template<typename Iterator>
84 class SynchronizedUniversalIterator: public SynchronizedIterator<Iterator> {
85 public:
86
87 /* "minimum" would be the smallest class bounded from below by all positions that appears in all OrdContainers.
88 * Are we sure that all positions at this class?
89 * Invariant: it can be true only if all positions are indeed synchronized.
90 */
91 bool synchronized_at_current_minimum = false;
92
93 /* advance() advances all positions to the NEXT minimum and returns true.
94 * (though the next minimum might be the current state if synchronized_at_current_minimum is false).
95 * or returns false if positions cannot be synchronized.
96 * If positions are synchronized to start with,
97 * then synchronized_at_current_minimum decides whether to stay or advance further.
98 * The general of the algorithm is to synchronize everybody with position[0].
99 */
100 bool advance()
101 {
102 //Nothing to synchronize.
103 if (this->positions.empty())
104 return false; //?? or not?
105
106 // If already synchronized, start moving forward by advancing position[0] (and so break synchronization).
107 if (this->synchronized_at_current_minimum) {
108 ++this->positions[0];
109 synchronized_at_current_minimum = false;
110 }
111
112 // If positions[0] has nowhere to go, then sync is not possible.
113 if (this->positions[0] == this->ends[0])
114 return false;
115
116 // Synchronise all positions with position[0].
117 for (size_t i = 1, positions_size = this->positions.size(); i < positions_size; ++i)
118 {
119 // positions[i]: ukazazel na zacatek (postupne se posunuje) moves z daneho stavu automatu i
120 // ends[i]: ukazazel na konec moves z daneho stavu automatu i
121 /*std::cout << "POS: aut0 (--[" << this->positions[0]->symbol << "]-->[";
122 for (auto a: this->positions[0]->states_to)
123 std::cout << a << ",";
124 std::cout << "]) aut" << i << " (--[" << this->positions[i]->symbol << "]-->[";
125 for (auto a: this->positions[i]->states_to)
126 std::cout << a << ",";
127 std::cout << "])\n";*/
128
129 // If some positions has nowhere to go, then sync is not possible.
130 if (this->positions[i] == this->ends[i])
131 return false;
132
133 // Advance position[i] and position[0] to the closest equal values.
134 while (*this->positions[i] != *this->positions[0]) {
135
136 // Advance position[i] to or beyond position[0].
137 while (*this->positions[i] < *this->positions[0]) {
138 ++this->positions[i];
139 if (this->positions[i] == this->ends[i])
140 return false;
141 }
142
143 // Advance position[0] to or beyond position[i].
144 while (*this->positions[i] > *this->positions[0]) {
145 ++this->positions[0];
146 if (this->positions[0] == this->ends[0])
147 return false;
148 }
149
150 // If position[0] changed, start from position 1 again.
151 // (note that
152 // i gets incremented at the end of the for-loop body,
153 // and that,
154 // since we are inside the for, there are at least two positions
155 // as the for starts with i=1.)
156 if (this->positions[0] > this->positions[1])
157 i=0;
158 }
159 }
160 this->synchronized_at_current_minimum = true;
161 return true;
162 }
163
164 // Returns the vector of current positions.
165 const std::vector<Iterator> get_current()
166 {
167 // How to return so that things don't get copied?
168 // Or maybe we should return a copy of positions anyway, to prevent a side effect of advance?
169 // Instead of returning a vector, we could also provide iterators through the result.
170 return this->positions;
171 };

176 SynchronizedIterator < Iterator > ::reset(size);
177 this->synchronized_at_current_minimum = false;

181 template<typename Iterator>
182 class SynchronizedExistentialIterator : public SynchronizedIterator<Iterator> {
183 public:
184
185 std::vector<Iterator> currently_synchronized; // Positions that are currently synchronized.
186
187 Iterator next_minimum; // the value we should synchronise on after the first next call of advance().
188
189 bool is_synchronized() {
190 return !currently_synchronized.empty();
191 }
192
193 const Iterator get_current_minimum()
194 {
195 if (currently_synchronized.empty())
196 throw std::runtime_error("Trying to get minimum from sync. ex. iterator which has no minimum. Don't do that ever again or your nose falls off!");
197 return currently_synchronized[0];
198 }
199
200
201 /* Advances all positions just above current_minimum,
202 * that is, to or above next_minimum.
203 * Those at next_minimum are added to currently_synchronized.
204 * Since next_minimum becomes the current minimum,
205 * new next_minimum must be updated too.
206 */
207 bool advance() {
208 // The next_minimum becomes the current current_minimum.
209 auto current_minimum = this->next_minimum;
210
211 // Here we collect the result.
212 currently_synchronized.clear();
213
214 for (size_t i = 0, positions_size = this->positions.size();i < positions_size;)
215 {
216 if (this->positions[i] == this->ends[i])
217 // If there is nothing left at the position, it is removed, that is,
218 // swapped with a position form the end of the vector,
219 // and shorten the vector.
220 // The same is done with ends.
221 while (this->positions[i] == this->ends[i] && i < positions_size)
222 {
223 this->positions[i] = this->positions[positions_size - 1];
224 this->ends[i] = this->ends [positions_size - 1];
225 this->positions.pop_back();
226 this->ends .pop_back();
227 positions_size--;
228 }
229 // If the i-th position, i.e., where we are now, was erased,
230 // then we reached the end of active positions.
231 if (i == positions_size)
232 return !currently_synchronized.empty();
233
234 // If we are at the current_minimum, then save it and advance the position.
235 if (*this->positions[i] == *current_minimum)
236 {
237 this->currently_synchronized.emplace_back(this->positions[i]);
238 ++this->positions[i];
239 continue;
240 }
241
242 // If the position (now larger than current_minimum) is smaller than next_minimum,
243 // or next_minimum has not yet been updated, then update the next_minimum.
244 if (*this->next_minimum > *this->positions[i] || *this->next_minimum == *current_minimum)
245 this->next_minimum = this->positions[i];
246
247 ++i;
248 }
249 return !currently_synchronized.empty();
250 };
251
252 /* Returns the vector of current still active positions.
253 * Beware, thy will be ordered differently from how there were input into the iterator.
254 * This is due to swapping of the emptied positions with positions at the end.
255 */
256 const std::vector<Iterator> get_current()
257 {
258 return this->currently_synchronized;
259 };
260
261 void push_back (const Iterator &begin, const Iterator &end) {
262
263 // Empty vector would not have any effect (unlike in the case of the universal iterator).
264 if (begin == end) return;
265
266 // Initialise next_minimum as the first position at the first vector.
267 if (this->positions.empty())
268 this->next_minimum = begin;
269
270
271 // If the first position is of the new vector is smaller than minimum,
272 // update minimum.
273 else if (*this->next_minimum > *begin)
274 this->next_minimum = begin;
275
276 // Let position point to the beginning the vector,
277 // save the end of the vector.
278 this->positions.emplace_back(begin);
279 this->ends.emplace_back(end);
280 };
281
282 explicit SynchronizedExistentialIterator(const int size=0) : SynchronizedIterator<Iterator>(size)
283 {
284 this->currently_synchronized.reserve(size);
285 };
286
287 void reset(const int size = 0) {
288 SynchronizedIterator < Iterator > ::reset(size);
289 if (size > 0) {
290 this->currently_synchronized.reserve(size);
291 }
292
293 this->currently_synchronized.clear();
294
295 };
296 };

298 /* In order to make initialisation of the sync. iterator nicer than inputting v.begin() and v.end()
299 * as the two parameters of the method push_back,
300 * this function wraps the method push_back,
301 * takes the iterator and v and extracts the v.begin() and v.end() from v.
302 */
303 template<class Container>
304 void push_back (SynchronizedIterator<typename Container::const_iterator> &i,const Container &container) {
305 i.push_back(container.begin(),container.end());
