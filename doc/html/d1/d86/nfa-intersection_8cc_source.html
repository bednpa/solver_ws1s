34void add_product_transition(Nfa& product, std::unordered_map<std::pair<State,State>, State>& product_map,
35 const std::pair<State,State>& pair_to_process,
36 Move& intersection_transition) {
37 if (intersection_transition.states_to.empty()) { return; }
38
39 auto& intersect_state_transitions{ product.transition_relation[product_map[pair_to_process]] };
40 auto symbol_transitions_iter{ intersect_state_transitions.find(intersection_transition) };
41
42 if (symbol_transitions_iter == intersect_state_transitions.end()) {
43 intersect_state_transitions.push_back(intersection_transition);
44 } else {
45 union_to_left(symbol_transitions_iter->states_to, intersection_transition.states_to);
46 }
47}

94Nfa intersection(const Nfa& const_lhs, const Nfa& const_rhs, bool preserve_epsilon,
95 std::unordered_map<std::pair<State,State>, State> *prod_map) {
96
97 Nfa lhs = const_lhs;
98 Nfa rhs = const_rhs;
99
100 Nfa product{}; // Product of the intersection.
101 // Product map for the generated intersection mapping original state pairs to new product states.
102 std::unordered_map<std::pair<State,State>, State> product_map{};
103 std::pair<State,State> pair_to_process{}; // State pair of original states currently being processed.
104 std::unordered_set<std::pair<State,State>> pairs_to_process{}; // Set of state pairs of original states to process.
105
106 // Initialize pairs to process with initial state pairs.
107 for (const State lhs_initial_state : lhs.initial_states) {
108 for (const State rhs_initial_state : rhs.initial_states) {
109 // Update product with initial state pairs.
110 const std::pair<State,State> this_and_other_initial_state_pair(lhs_initial_state, rhs_initial_state);
111 const State new_intersection_state = product.add_state();
112
113 product_map[this_and_other_initial_state_pair] = new_intersection_state;
114 pairs_to_process.insert(this_and_other_initial_state_pair);
115
116 product.initial_states.push_back(new_intersection_state);
117 if (lhs.has_final(lhs_initial_state) && rhs.has_final(rhs_initial_state)) {
118 product.final_states.push_back(new_intersection_state);
119 }
120 }
121 }
122
123 Nfa rhs_copy = rhs;
124 Nfa lhs_copy = lhs;
125 std::vector<std::pair<State,State>> end_inner_states_lhs;
126 std::vector<std::pair<State,State>> end_inner_states_rhs;
127
128 while (!pairs_to_process.empty()) {
129 pair_to_process = *pairs_to_process.cbegin();
130 pairs_to_process.erase(pair_to_process);
131
132 // Compute classic product for current state pair.
133
134 Mata::Util::SynchronizedUniversalIterator<Mata::Util::OrdVector<Move>::const_iterator> sync_iterator(2);
135
136 Moves act_moves_lhs = lhs_copy.transition_relation[pair_to_process.first];
137 Moves act_moves_rhs = rhs_copy.transition_relation[pair_to_process.second];
138
139 Mata::Util::push_back(sync_iterator,act_moves_lhs);
140 Mata::Util::push_back(sync_iterator,act_moves_rhs);
141
142 while (sync_iterator.advance()) {
143 std::vector<Moves::const_iterator> moves = sync_iterator.get_current();
144 assert(moves.size() == 2); // One move per state in the pair.
145
146 // Compute product for state transitions with same symbols.
147 // Find all transitions that have the same symbol for first and the second state in the pair_to_process.
148 // Create transition from the pair_to_process to all pairs between states to which first transition goes
149 // and states to which second one goes.
150 Move intersection_transition{moves[0]->symbol};
151 for (State this_state_to: moves[0]->states_to)
152 {
153 for (State other_state_to: moves[1]->states_to)
154 {
155 // A --> B
156 if (this_state_to.jump < other_state_to.jump) {
157
158 // remove A --> B
159 rhs_copy.remove_trans(pair_to_process.second, intersection_transition.symbol, other_state_to);
160
161 // create inner state
162 State inner_state;
163 auto it = std::find_if(end_inner_states_rhs.begin(), end_inner_states_rhs.end(), [&this_state_to,&other_state_to](const auto &pair) { return pair.first.state == other_state_to.state && pair.first.jump - pair.second.jump == other_state_to.jump - this_state_to.jump && other_state_to.jump - this_state_to.jump > 0; });
164
165 if (it != end_inner_states_rhs.end()) {
166 inner_state.state = it->second.state;
167 inner_state.jump = this_state_to.jump;
168 } else {
169 inner_state.state = rhs_copy.add_state();
170 inner_state.jump = this_state_to.jump;
171 end_inner_states_rhs.push_back(std::pair<State,State>(other_state_to, inner_state));
172
173 // add X --> B
174 rhs_copy.add_trans(inner_state, 0, State{other_state_to.state, other_state_to.jump - this_state_to.jump});
175 rhs_copy.add_trans(inner_state, 1, State{other_state_to.state, other_state_to.jump - this_state_to.jump});
176 }
177
178 // add A --> X
179 rhs_copy.add_trans(pair_to_process.second, intersection_transition.symbol, inner_state);
180
181 // set B = X
182 other_state_to = inner_state;
183 }
184
185 // A --> B
186 if (this_state_to.jump > other_state_to.jump) {
187
188 // remove A --> B
189 lhs_copy.remove_trans(pair_to_process.first, intersection_transition.symbol, this_state_to);
190
191 // create inner state
192 State inner_state;
193 auto it = std::find_if(end_inner_states_lhs.begin(), end_inner_states_lhs.end(), [&this_state_to,&other_state_to](const auto &pair) { return pair.first.state == this_state_to.state && pair.first.jump - pair.second.jump == this_state_to.jump - other_state_to.jump && this_state_to.jump - other_state_to.jump > 0; });
194
195 if (it != end_inner_states_lhs.end()) {
196 inner_state.state = it->second.state;
197 inner_state.jump = other_state_to.jump;
198 } else {
199 inner_state.state = lhs_copy.add_state();
200 inner_state.jump = other_state_to.jump;
201 end_inner_states_lhs.push_back(std::pair<State,State>(this_state_to, inner_state));
202
203 // add X --> B
204 lhs_copy.add_trans(inner_state, 0, State{this_state_to.state, this_state_to.jump - other_state_to.jump});
205 lhs_copy.add_trans(inner_state, 1, State{this_state_to.state, this_state_to.jump - other_state_to.jump});
206 }
207
208 // add A --> X
209 lhs_copy.add_trans(pair_to_process.first, intersection_transition.symbol, inner_state);
210
211 // set B = X
212 this_state_to = inner_state;
213 }
214 create_product_state_and_trans(
215 product, product_map, lhs_copy, rhs_copy, pairs_to_process,
216 this_state_to, other_state_to, intersection_transition
217 );
218 }
219 }
220 add_product_transition(product, product_map, pair_to_process, intersection_transition);
221 }
222 }
223
224 if (prod_map != nullptr) { *prod_map = product_map; }
225
226 return product;
227} // intersection().
